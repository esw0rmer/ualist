--==[ Guide
>>> from ualist import ua
>>> print ua() or print ua(argument)

--==[ Argument
~> ua() = Random
~> ua(win) = Windows
~> ua(lin) = Linux
~> ua(mac) = Macintosh
~> ua(and) = Android
~> ua(mob) = Mobile
~> ua(tab) = Tablet

>>> ualist.ua()
'Mozilla/5.0 (Tablet; rv:18.1) Gecko/18.1 Firefox/18.1'
>>> ualist.ua("mac")
'Mozilla/5.0 (Macintosh; U; Intel Mac OS X 10.5; ko; rv:1.9.1b2) Gecko/20081201 Firefox/3.1b2'
